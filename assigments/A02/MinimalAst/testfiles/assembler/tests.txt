--gcd-test
int gcd1(int a, int b) {
  while (a != b) {
    if (a > b) {
      a = a - b;
    } else {
      b = b - a;
    }
  }
  return a;
}
int gcd2(int a, int b) {
  if (b == 0) {
    return a;
  }
  return gcd2(b, a % b);
}
int main() {
  int a;
  int b;
  a = read();
  b = read();
  print(gcd1(a, b));
  print(gcd2(a, b));
  int diff = gcd1(a, b) - gcd2(a, b);
  print(diff);
  return 0;
}

5
